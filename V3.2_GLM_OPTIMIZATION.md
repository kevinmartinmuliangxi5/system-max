class GLMCodeReviewer:
    """基于GLM的代码审查系统"""

    def __init__(self):
        self.client = anthropic.Anthropic(
            api_key="434698bbe0754202aabea54a1b6d184d.zPhgfbdAFrX46GDg",
            base_url="https://open.bigmodel.cn/api/anthropic"
        )
        self.ce_agent = GLMCompoundEngineeringAgent()

    def review_phase_output(self, phase: Dict, output_files: List[str]) -> Dict:
        """完整的Phase审查"""

        results = {
            "phase": phase["task_name"],
            "files_reviewed": len(output_files),
            "checks": {},
            "overall_score": 0,
            "approved": False
        }

        # 1. 静态检查（快速，免费）
        static_results = self._static_analysis(output_files)
        results["checks"]["static_analysis"] = static_results

        # 如果静态分析发现严重问题，先返回
        if static_results["critical_issues"] > 0:
            results["approved"] = False
            results["reason"] = "静态分析发现严重问题"
            return results

        # 2. GLM深度审查（使用包月额度）
        for file_path in output_files:
            if file_path.endswith(('.py', '.js', '.ts', '.java')):
                code_content = self._read_file(file_path)
                review = self.ce_agent.code_reviewer_agent(
                    code=code_content,
                    context=phase["instruction"]
                )
                results["checks"][file_path] = review

                # 如果发现高危问题，拒绝
                high_severity = [
                    issue for issue in review["issues"]
                    if issue["severity"] == "high"
                ]
                if high_severity:
                    results["approved"] = False
                    results["high_severity_issues"] = high_severity
                    return results

        # 3. 架构一致性检查（使用GLM）
        arch_check = self._check_architecture_consistency(
            output_files, phase
        )
        results["checks"]["architecture"] = arch_check

        # 计算总分
        all_scores = [
            check["quality_score"]
            for check in results["checks"].values()
            if isinstance(check, dict) and "quality_score" in check
        ]
        results["overall_score"] = sum(all_scores) / len(all_scores) if all_scores else 0

        # 决定是否通过
        results["approved"] = results["overall_score"] >= 7.0

        return results

    def _static_analysis(self, files: List[str]) -> Dict:
        """静态分析（免费工具）"""
        # pylint, bandit等
        # 这部分免费，快速
        pass

    def _check_architecture_consistency(self, files: List[str], phase: Dict) -> Dict:
        """架构一致性检查（用GLM）"""

        prompt = f"""你是架构审查专家。

任务：{phase["task_name"]}
产出文件：{', '.join(files)}

请检查：
1. 是否符合项目架构规范
2. 模块间依赖是否合理
3. 接口设计是否一致
4. 是否有架构债务

返回JSON：
{{
    "quality_score": 0-10,
    "issues": [],
    "suggestions": []
}}
"""

        response = self.client.messages.create(
            model="glm-4.7",
            max_tokens=2000,
            messages=[{"role": "user", "content": prompt}]
        )

        return json.loads(response.content[0].text)

### 优化5: 智能模型选择（GLM-4.7 vs GLM-4.5-Air）

**策略**:
```python
class SmartModelSelector:
    """智能选择GLM模型"""

    def __init__(self):
        self.models = {
            "glm-4.7": {
                "capability": "high",
                "speed": "medium",
                "use_for": ["complex_analysis", "code_review", "architecture"]
            },
            "glm-4.5-air": {
                "capability": "medium",
                "speed": "fast",
                "use_for": ["simple_questions", "quick_checks", "formatting"]
            }
        }

    def select_model(self, task_type: str, task_complexity: str) -> str:
        """选择合适的模型"""

        # 复杂任务用GLM-4.7
        if task_complexity in ["high", "critical"]:
            return "glm-4.7"

        # 简单任务用GLM-4.5-Air（更快）
        if task_type in ["format_check", "simple_question", "quick_review"]:
            return "glm-4.5-air"

        # 默认用GLM-4.7（反正包月）
        return "glm-4.7"
```
