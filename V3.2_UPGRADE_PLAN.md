class StaticAnalysisReviewer:
    """静态分析审查器 - 完全免费"""

        self.tools = {

        results = {

class LocalAIReviewer:
    """本地AI代码审查 - 免费"""

#### 第三层: 云端AI审查（按需使用）⚠️

**智能降级策略**:

```python
class HybridReviewer:
    """混合审查器 - 成本优化"""

    def __init__(self):
        self.static = StaticAnalysisReviewer()  # 免费
        self.local_ai = LocalAIReviewer()       # 免费
        self.cloud_ai = GLMReviewer()           # 付费但便宜

        # 成本阈值
        self.use_cloud_threshold = {
            'critical_security': True,   # 安全问题必须用云
            'architecture': False,       # 架构用本地
            'simple_logic': False,       # 简单逻辑用本地
        }

    def review(self, code: str, file_type: str,
               importance: str) -> Dict:
        """多层审查策略"""

        # 第一层：静态分析（总是执行，免费）
        static_result = self.static.review_file(code)

        # 如果静态分析发现严重问题，直接返回
        if static_result['quality_score'] < 5:
            return {
                'source': 'static',
                'cost': 0,
                'issues': static_result['issues']
            }

        # 第二层：本地AI（默认，免费）
        local_result = self.local_ai.review_code(code)

        # 第三层：云端AI（仅关键情况）
        if importance == 'critical' or \
           'security' in local_result['tags']:
            cloud_result = self.cloud_ai.review_code(code)
            return {
                'source': 'cloud',
                'cost': 0.001,  # 约0.001元
                'issues': cloud_result['issues']
            }

        return {
            'source': 'local',
            'cost': 0,
            'issues': local_result['issues']
        }
```

```python
# .ralph/tools/cost_controller.py
class CostController:
    """成本控制器"""

    def __init__(self):
        self.budget = {
            'daily_limit': 5.0,      # 每天5元
            'per_task_limit': 0.5,   # 每任务0.5元
            'warning_threshold': 0.8 # 80%预警
        }

        self.costs = {
            'static_analysis': 0,
            'local_ai': 0,
            'glm_api': 0.001,        # 每次0.001元
            'claude_sonnet': 0.015,  # 每次0.015元
            'claude_opus': 0.075,    # 每次0.075元
        }

        self.usage_today = 0

    def should_use_cloud_api(self, task_type: str,
                           task_importance: str) -> str:
        """决定使用哪个API"""

        # 检查预算
        if self.usage_today >= self.budget['daily_limit']:
            return 'local'  # 超预算，用本地

        # 根据重要性决定
        if task_importance == 'critical':
            if self.usage_today < self.budget['daily_limit'] * 0.5:
                return 'glm'  # 还有预算，用便宜的GLM
            else:
                return 'local'

        # 默认本地
        return 'local'

    def record_usage(self, api_type: str):
        """记录使用量"""
        cost = self.costs.get(api_type, 0)
        self.usage_today += cost

        # 预警
        if self.usage_today >= self.budget['daily_limit'] * 0.8:
            print(f"⚠️  成本预警: 今日已使用 {self.usage_today}元")

    def get_cost_report(self) -> Dict:
        """成本报告"""
        return {
            'today_spent': self.usage_today,
            'daily_limit': self.budget['daily_limit'],
            'remaining': self.budget['daily_limit'] - self.usage_today,
            'usage_percentage': self.usage_today / self.budget['daily_limit']
        }
```
